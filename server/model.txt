Meteor.methods({
	save_to_report: function(time, state, identifier) {
		var today = new Date(moment().format('MMM D, YYYY'));
		return Report.update(
			{
				date: today,
				state: state,
				identifier: identifier
			},
			{
				$inc: { time: time }
			},
			{
				upsert: true
			}					
		);		
	},
	update_activities: function(insert_id, state, identifier ) {
	
		var now = Math.round(new Date().getTime() / 1000);

		if(insert_id) {
			var lastEntry = Activities.findOne({_id: insert_id});
			
			if(lastEntry) {
				if (lastEntry.state == state) {
					return;
				}
				
				var time_spent = now - lastEntry.created_at;

				if (time_spent < 1) {
					// only update state w/ the new one, leaving creation date the same
					var upd = Activities.update(lastEntry._id,
						{
							$set:
							{state: state}
						});
					console.log('updated');
					return;
					
				} else {
					
				
					//update anstatt remove
					Activities.remove(lastEntry._id);
					Meteor.call("save_to_report", time_spent, lastEntry.state, identifier, function() {
						console.log('Saved to server ' + time_spent);
					});
				}
			}		
		}
	
		return Activities.insert({
				created_at: now,
				state:      state,
				identifier:  identifier
		});	
	}	
});
